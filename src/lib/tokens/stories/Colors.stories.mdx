import {
  Meta,
  Source,
  ColorPalette,
  ColorItem,
} from '@storybook/addon-docs/blocks';
import colors from '../colors.js';

<Meta
  title='Styleguide/Colors'
  parameters={{ previewTabs: { canvas: { hidden: true } } }}
/>

# Colors

## Usage

<Source
  language='css'
  code={`
.someclass {
  color: var(--color-primary-500);
}
  `}
/>

<div>
  {Object.keys(colors).map((key) => {
    return (
      <div key={`${key}`}>
        <h2>{key}</h2>
        <ColorPalette>
          {Object.keys(colors[key]).map((colorName) => {
            return (
              <ColorItem
                key={`${key}-${colorName}`}
                title={`color-${key}-${colorName}`}
                colors={[colors[key][colorName]]}
              />
            );
          })}
        </ColorPalette>
      </div>
    );
  }, [])}
</div>
