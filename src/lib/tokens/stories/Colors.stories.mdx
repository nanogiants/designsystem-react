import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks';
import tokens from '../index.js';

<Meta title="Styleguide/Colors" />

# Colors

<ColorPalette>
  {Object.keys(tokens.colors).reduce((acc, key) => {
    acc.push(
      ...Object.keys(tokens.colors[key]).map((colorName) => {
        return (
          <ColorItem
            key={`${key}-${colorName}`}
            title={`${key}-${colorName}`}
            colors={[tokens.colors[key][colorName]]}
          />
        );
      })
    );
    return acc;
  }, [])}
</ColorPalette>
